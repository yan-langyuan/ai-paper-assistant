## 五、阶段二：数据库与基础设施

### 5.1 数据库表设计（对应 PRD 4个模块）

```sql
-- MySQL 8.0+

-- 用户表
CREATE TABLE `user` (
    `id` BIGINT AUTO_INCREMENT PRIMARY KEY,
    `username` VARCHAR(50) NOT NULL UNIQUE,
    `password` VARCHAR(255) NOT NULL,
    `email` VARCHAR(100),
    `grade` VARCHAR(20) COMMENT '年级',
    `major` VARCHAR(50) COMMENT '专业',
    `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
    `updated_at` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 文献表（核心）
CREATE TABLE `literature` (
    `id` BIGINT AUTO_INCREMENT PRIMARY KEY,
    `user_id` BIGINT NOT NULL,
    `title` VARCHAR(500) NOT NULL COMMENT '文献标题',
    `authors` VARCHAR(500) COMMENT '作者（逗号分隔）',
    `journal` VARCHAR(300) COMMENT '期刊/会议名称',
    `volume` VARCHAR(20) COMMENT '卷号',
    `issue` VARCHAR(20) COMMENT '期号',
    `pages` VARCHAR(50) COMMENT '页码范围',
    `year` INT COMMENT '发表年份',
    `doi` VARCHAR(200) COMMENT 'DOI',
    `abstract_text` TEXT COMMENT '原文摘要',
    `full_text` LONGTEXT COMMENT '全文内容（PDF解析结果）',
    `ai_summary` JSON COMMENT 'AI结构化摘要(JSON)',
    `status` VARCHAR(20) DEFAULT 'PENDING' COMMENT 'PENDING/PARSING/COMPLETED/FAILED',
    `file_path` VARCHAR(500) COMMENT '上传文件存储路径',
    `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
    `updated_at` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (`user_id`) REFERENCES `user`(`id`) ON DELETE CASCADE,
    INDEX `idx_user_id` (`user_id`),
    INDEX `idx_status` (`status`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 改写记录表（降重闭环）
CREATE TABLE `rewrite_record` (
    `id` BIGINT AUTO_INCREMENT PRIMARY KEY,
    `user_id` BIGINT NOT NULL,
    `original_text` TEXT NOT NULL COMMENT '原始文本',
    `similarity` DECIMAL(5,2) COMMENT '原始相似度%',
    `rewritten_text` TEXT COMMENT '改写后文本',
    `scheme_type` VARCHAR(10) COMMENT 'A(保守)/B(均衡)/C(激进)',
    `new_similarity` DECIMAL(5,2) COMMENT '改写后相似度%',
    `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (`user_id`) REFERENCES `user`(`id`) ON DELETE CASCADE,
    INDEX `idx_user_id` (`user_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- AI使用计数表
CREATE TABLE `ai_usage_log` (
    `id` BIGINT AUTO_INCREMENT PRIMARY KEY,
    `user_id` BIGINT NOT NULL,
    `action_type` VARCHAR(50) COMMENT 'SUMMARY/REWRITE/DETECT',
    `char_count` INT COMMENT '处理字数',
    `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (`user_id`) REFERENCES `user`(`id`) ON DELETE CASCADE,
    INDEX `idx_user_date` (`user_id`, `created_at`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

### 5.3 CORS 与安全配置

```java
// CorsConfig.java — 允许前端跨域访问
@Configuration
public class CorsConfig implements WebMvcConfigurer {
    @Override
    public void addCorsMappings(CorsRegistry registry) {
        registry.addMapping("/api/**")
            .allowedOrigins("http://localhost:8081")
            .allowedMethods("GET", "POST", "PUT", "DELETE")
            .allowedHeaders("*")
            .allowCredentials(true);
    }
}

// SecurityConfig.java — JWT 认证（MVP阶段简化）
@Configuration
@EnableWebSecurity
public class SecurityConfig {
    @Bean
    public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
        http
            .cors(Customizer.withDefaults())
            .csrf(AbstractHttpConfigurer::disable)
            .authorizeHttpRequests(auth -> auth
                .requestMatchers("/api/auth/**").permitAll()
                .requestMatchers("/api/**").authenticated()
                .anyRequest().permitAll()
            );
        return http.build();
    }
}
```

#### useAiCounter.js（AI计数器 Composable — F06）

```javascript
// src/composables/useAiCounter.js
// 核心：AI使用量前端本地计数，触发三档提醒
import { ref, computed, watch } from 'vue';

const STORAGE_KEY = 'ai_usage_counter';

export function useAiCounter(totalDocWords = ref(5000)) {
  const aiWords = ref(loadFromStorage());

  function loadFromStorage() {
    const stored = localStorage.getItem(STORAGE_KEY);
    return stored ? JSON.parse(stored) : { daily: 0, weekly: 0, docTotal: 0 };
  }

  function addUsage(charCount) {
    aiWords.value.daily += charCount;
    aiWords.value.weekly += charCount;
    aiWords.value.docTotal += charCount;
    localStorage.setItem(STORAGE_KEY, JSON.stringify(aiWords.value));
  }

  const ratio = computed(() => {
    if (totalDocWords.value === 0) return 0;
    return (aiWords.value.docTotal / totalDocWords.value) * 100;
  });

  // 三档提醒阈值（PRD F06）
  const warningLevel = computed(() => {
    const r = ratio.value;
    if (r >= 50) return 'high';    // 弹窗强制确认
    if (r >= 30) return 'medium';  // 弹窗温和提醒
    if (r >= 20) return 'low';     // 侧边栏浅黄
    return null;
  });

  return { aiWords, ratio, warningLevel, addUsage };
}
```
